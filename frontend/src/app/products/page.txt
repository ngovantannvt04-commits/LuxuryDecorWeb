"use client";

import Header from "@/components/layout/Header";
import Footer from "@/components/layout/Footer";
import { useEffect, useState, useCallback } from "react"; // Th√™m useCallback
import { productService } from "@/services/product.service";
import { Category, Product } from "@/types/product.types";
import Image from "next/image";
import { Filter, Search } from "lucide-react";
// üëá Import hook c·ªßa Next.js ƒë·ªÉ thao t√°c URL
import { useRouter, useSearchParams } from "next/navigation"; 

export default function ProductsPage() {
  const router = useRouter();
  const searchParams = useSearchParams();

  // 1. L·∫•y gi√° tr·ªã t·ª´ URL xu·ªëng (N·∫øu kh√¥ng c√≥ th√¨ l·∫•y m·∫∑c ƒë·ªãnh)
  // Backend trang b·∫Øt ƒë·∫ßu t·ª´ 0, nh∆∞ng tr√™n URL ta n√™n ƒë·ªÉ t·ª´ 1 cho th√¢n thi·ªán user
  const currentPage = Number(searchParams.get("page")) || 1; 
  const currentKeyword = searchParams.get("keyword") || "";
  const currentCat = searchParams.get("categoryId") ? Number(searchParams.get("categoryId")) : undefined;
  const currentMinPrice = searchParams.get("minPrice") ? Number(searchParams.get("minPrice")) : undefined;
  const currentMaxPrice = searchParams.get("maxPrice") ? Number(searchParams.get("maxPrice")) : undefined;
  const currentSort = searchParams.get("sortBy") || "newest";

  // State ch·ªâ d√πng ƒë·ªÉ l∆∞u D·ªØ li·ªáu hi·ªÉn th·ªã
  const [products, setProducts] = useState<Product[]>([]);
  const [categories, setCategories] = useState<Category[]>([]);
  const [loading, setLoading] = useState(true);
  const [totalPages, setTotalPages] = useState(0);

  // State t·∫°m ƒë·ªÉ g√µ input t√¨m ki·∫øm (tr√°nh push URL li√™n t·ª•c m·ªói khi g√µ 1 ch·ªØ)
  const [tempKeyword, setTempKeyword] = useState(currentKeyword);

  // Load danh m·ª•c 1 l·∫ßn ƒë·∫ßu
  useEffect(() => {
    productService.getCategories().then(setCategories).catch(console.error);
  }, []);

  // ƒê·ªìng b·ªô tempKeyword n·∫øu URL thay ƒë·ªïi (tr∆∞·ªùng h·ª£p user back/forward tr√¨nh duy·ªát)
  useEffect(() => {
    setTempKeyword(currentKeyword);
  }, [currentKeyword]);

  // === 2. H√ÄM CHUNG ƒê·ªÇ UPDATE URL ===
  // H√†m n√†y s·∫Ω gi·ªØ l·∫°i c√°c param c≈©, ch·ªâ thay ƒë·ªïi c√°i c·∫ßn thi·∫øt
  const updateUrl = useCallback((newParams: Record<string, string | number | undefined>) => {
    const params = new URLSearchParams(searchParams.toString());

    Object.keys(newParams).forEach(key => {
      const value = newParams[key];
      if (value === undefined || value === "" || value === 0) {
        params.delete(key);
      } else {
        params.set(key, String(value));
      }
    });

    // Lu√¥n reset v·ªÅ trang 1 n·∫øu thay ƒë·ªïi b·ªô l·ªçc (tr·ª´ khi ƒëang ƒë·ªïi page)
    if (!newParams.page && newParams.page !== 0) {
        params.set("page", "1");
    }

    router.push(`/products?${params.toString()}`);
  }, [router, searchParams]);


  // === 3. G·ªåI API KHI URL THAY ƒê·ªîI ===
  useEffect(() => {
    const fetchProducts = async () => {
      setLoading(true);
      try {
        const res = await productService.getAllProducts({
          page: currentPage - 1, // URL l√† trang 1, nh∆∞ng Backend c·∫ßn trang 0
          size: 12,
          keyword: currentKeyword,
          categoryId: currentCat,
          minPrice: currentMinPrice,
          maxPrice: currentMaxPrice,
          sortBy: currentSort
        });
        setProducts(res.content);
        setTotalPages(res.totalPages);
      } catch (error) {
        console.error(error);
      } finally {
        setLoading(false);
      }
    };

    fetchProducts();
    // Ch·∫°y l·∫°i m·ªói khi params tr√™n URL thay ƒë·ªïi
  }, [currentPage, currentKeyword, currentCat, currentMinPrice, currentMaxPrice, currentSort]);


  // === 4. C√ÅC H√ÄM X·ª¨ L√ù S·ª∞ KI·ªÜN ===
  
  // X·ª≠ l√Ω t√¨m ki·∫øm (Debounce ho·∫∑c ·∫•n Enter)
  const handleSearchSubmit = () => {
      updateUrl({ keyword: tempKeyword, categoryId: undefined }); // T√¨m ki·∫øm th√¨ b·ªè ch·ªçn danh m·ª•c
  };

  const handleCategoryClick = (catId?: number) => {
      // Ch·ªçn danh m·ª•c th√¨ b·ªè t·ª´ kh√≥a t√¨m ki·∫øm
      updateUrl({ categoryId: catId, keyword: undefined });
  };

  const handlePageChange = (newPage: number) => {
      updateUrl({ page: newPage }); // newPage n√†y l√† s·ªë hi·ªÉn th·ªã (1,2,3)
  };

  const handlePriceApply = (min?: number, max?: number) => {
      updateUrl({ minPrice: min, maxPrice: max });
  };

  const formatCurrency = (amount: number) => 
     new Intl.NumberFormat('vi-VN', { style: 'currency', currency: 'VND' }).format(amount);

  return (
    <div className="bg-gray-50 min-h-screen">
      <Header />
      
      <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-10">
        <div className="flex flex-col md:flex-row gap-8">
          
          <aside className="w-full md:w-64 space-y-8 h-fit">
            {/* Input T√¨m ki·∫øm */}
            <div className="bg-white p-5 rounded-xl shadow-sm">
                <h3 className="font-bold mb-3 flex items-center gap-2"><Search size={18}/> T√¨m ki·∫øm</h3>
                <div className="flex gap-2">
                    <input 
                        type="text" 
                        placeholder="T√™n s·∫£n ph·∫©m..." 
                        className="w-full border border-gray-200 rounded-lg p-2 focus:outline-none focus:ring-1 focus:ring-black"
                        value={tempKeyword}
                        onChange={(e) => setTempKeyword(e.target.value)}
                        onKeyDown={(e) => e.key === 'Enter' && handleSearchSubmit()}
                    />
                    <button onClick={handleSearchSubmit} className="bg-black text-white p-2 rounded-lg">
                        <Search size={16}/>
                    </button>
                </div>
            </div>

            {/* Danh m·ª•c - C·∫≠p nh·∫≠t logic active */}
            <div className="bg-white p-5 rounded-xl shadow-sm">
                <h3 className="font-bold mb-3 flex items-center gap-2"><Filter size={18}/> Danh m·ª•c</h3>
                <ul className="space-y-2">
                    <li 
                        className={`cursor-pointer hover:text-amber-600 ${!currentCat ? 'font-bold text-amber-600' : 'text-gray-600'}`}
                        onClick={() => handleCategoryClick(undefined)}
                    >
                        T·∫•t c·∫£
                    </li>
                    {categories.map((cat) => (
                        <li 
                            key={cat.categoryId}
                            className={`cursor-pointer hover:text-amber-600 ${currentCat === cat.categoryId ? 'font-bold text-amber-600' : 'text-gray-600'}`}
                            onClick={() => handleCategoryClick(cat.categoryId)}
                        >
                            {cat.categoryName}
                        </li>
                    ))}
                </ul>
            </div>

            {/* Kho·∫£ng gi√° */}
            <div className="bg-white p-5 rounded-xl shadow-sm">
                <h3 className="font-bold mb-3">Kho·∫£ng gi√°</h3>
                {/* Ta d√πng uncontrolled component (ref) ho·∫∑c state t·∫°m cho input gi√° c≈©ng ƒë∆∞·ª£c, ·ªü ƒë√¢y d√πng t·∫°m state c·ª•c b·ªô */}
                <PriceFilter onApply={handlePriceApply} initialMin={currentMinPrice} initialMax={currentMaxPrice} />
            </div>
          </aside>

          <main className="flex-1">
            {/* Sort Header */}
            <div className="flex justify-between items-center mb-6">
                <h2 className="text-2xl font-serif font-bold">
                    {currentKeyword ? `T√¨m ki·∫øm: "${currentKeyword}"` : (currentCat ? categories.find(c => c.categoryId === currentCat)?.categoryName : "T·∫•t c·∫£ s·∫£n ph·∫©m")}
                </h2>
                <select 
                    className="border border-gray-300 rounded-lg p-2 bg-white cursor-pointer focus:outline-none"
                    value={currentSort}
                    onChange={(e) => updateUrl({ sortBy: e.target.value })}
                >
                    <option value="newest">M·ªõi nh·∫•t</option>
                    <option value="price_asc">Gi√°: Th·∫•p ƒë·∫øn Cao</option>
                    <option value="price_desc">Gi√°: Cao ƒë·∫øn Th·∫•p</option>
                </select>
            </div>

            {/* Products Grid (Gi·ªØ nguy√™n ph·∫ßn hi·ªÉn th·ªã) */}
            {loading ? (
                 <div className="h-64 flex items-center justify-center">ƒêang t·∫£i d·ªØ li·ªáu...</div>
            ) : products.length === 0 ? (
                 <div className="h-64 flex flex-col items-center justify-center text-gray-500">
                    <p>Kh√¥ng t√¨m th·∫•y s·∫£n ph·∫©m n√†o ph√π h·ª£p.</p>
                 </div>
            ) : (
                <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-6">
                    {products.map((product) => (
                        <div key={product.productId} className="bg-white p-4 rounded-xl shadow-sm hover:shadow-md transition group">
                             {/* ... Ph·∫ßn hi·ªÉn th·ªã Card s·∫£n ph·∫©m (Gi·ªØ nguy√™n code c≈©) ... */}
                             <div className="relative h-64 w-full overflow-hidden rounded-lg bg-gray-100 mb-4">
                                <Image 
                                    src={product.image || "https://placehold.co/400x400"} 
                                    alt={product.productName} 
                                    fill 
                                    className="object-cover group-hover:scale-105 transition duration-500" 
                                />
                             </div>
                             <div className="text-xs text-gray-500 mb-1">{product.categoryName}</div>
                             <h3 className="font-bold text-gray-900 truncate">{product.productName}</h3>
                             <div className="flex justify-between items-center mt-2">
                                <span className="text-amber-700 font-bold">{formatCurrency(product.price)}</span>
                             </div>
                        </div>
                    ))}
                </div>
            )}

            {/* Pagination Controls - S·ª≠a logic hi·ªÉn th·ªã page */}
            {totalPages > 1 && (
                <div className="flex justify-center mt-10 gap-2">
                    <button 
                        disabled={currentPage === 1}
                        onClick={() => handlePageChange(currentPage - 1)}
                        className="px-4 py-2 border rounded-lg hover:bg-gray-100 disabled:opacity-50"
                    >
                        Tr∆∞·ªõc
                    </button>
                    <span className="px-4 py-2 bg-black text-white rounded-lg">
                        Trang {currentPage} / {totalPages}
                    </span>
                    <button 
                        disabled={currentPage === totalPages}
                        onClick={() => handlePageChange(currentPage + 1)}
                        className="px-4 py-2 border rounded-lg hover:bg-gray-100 disabled:opacity-50"
                    >
                        Sau
                    </button>
                </div>
            )}
          </main>
        </div>
      </div>
      <Footer />
    </div>
  );
}

// Component ph·ª• ƒë·ªÉ x·ª≠ l√Ω input gi√° (tr√°nh render l·∫°i c·∫£ trang khi g√µ t·ª´ng s·ªë)
function PriceFilter({ onApply, initialMin, initialMax }: { onApply: (min?: number, max?: number) => void, initialMin?: number, initialMax?: number }) {
    const [min, setMin] = useState<number | undefined>(initialMin);
    const [max, setMax] = useState<number | undefined>(initialMax);

    return (
        <>
            <div className="flex gap-2 mb-2">
                <input 
                    type="number" placeholder="Min" className="w-1/2 border p-2 rounded text-sm"
                    value={min || ''} onChange={(e) => setMin(Number(e.target.value) || undefined)}
                />
                <input 
                    type="number" placeholder="Max" className="w-1/2 border p-2 rounded text-sm"
                    value={max || ''} onChange={(e) => setMax(Number(e.target.value) || undefined)}
                />
            </div>
            <button 
                onClick={() => onApply(min, max)}
                className="w-full bg-black text-white py-2 rounded-lg text-sm hover:bg-gray-800"
            >
                √Åp d·ª•ng
            </button>
        </>
    )
}
